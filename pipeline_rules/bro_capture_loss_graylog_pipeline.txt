rule "Extract bro_capture_loss log fields"
when
  has_field("application_name") &&
  contains(value: to_string($message.application_name), search: "bro_capture_loss", ignore_case: true)
then
  let m = split("\\|", to_string($message.message));

  //set_field("<field>", m[<id_num>]);
  set_field("ts", m[0]);
  set_field("ts_delta", m[1]);
  set_field("peer", m[2]);
  set_field("gaps", to_long(m[3]));
  set_field("acks", to_long(m[4]));
  set_field("percent_lost", to_long(m[5]));
end
